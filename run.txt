dependencies needed:
rust
node
npm

# Install Circom
git clone https://github.com/iden3/circom.git
cd circom && cargo build --release
cargo install --path circom

# Install snarkjs globally
npm install -g snarkjs

# Install circomlib in your circuits folder 
npm install 

circom VoteProof.circom --r1cs --wasm --sym -l node_modules

# Download Powers of Tau (phase 1) - use pot20 since your circuit is large
wget https://hermez.s3-eu-west-1.amazonaws.com/powersOfTau28_hez_final_20.ptau

# Phase 2 (circuit specific)
snarkjs groth16 setup VoteProof.r1cs powersOfTau28_hez_final_20.ptau circuit_0000.zkey

# Contribute randomness (in production, do this properly with multiple parties)
snarkjs zkey contribute circuit_0000.zkey circuit_final.zkey --name="contribution" -v

# Export verification key
snarkjs zkey export verificationkey circuit_final.zkey verification_key.json
